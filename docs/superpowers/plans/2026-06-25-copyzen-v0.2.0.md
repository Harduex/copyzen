# copyzen v0.2.0 Implementation Plan — image thumbnails + delete/clear

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship two picker features together as v0.2.0 — (a) image entries render as fuzzel thumbnails and paste back as real images; (b) `Ctrl+D` deletes a record and `Ctrl+Shift+C` clears history, each behind a two-step confirmation.

**Architecture:** copyzen stays a byte-faithful bbolt store driven by pipes; fuzzel is the UI via `scripts/copyzen-menu`. Image support is a pure display + paste-path change (sniff magic bytes, cache downscaled PNG thumbnails, expose the MIME so the menu can `wl-copy --type`) — no store-schema change. Delete/clear reuse the existing `copyzen delete`/`wipe` commands; the new work is fuzzel keybindings plus a confirm-state machine in the menu wrapper.

**Tech Stack:** Go 1.23, `go.etcd.io/bbolt`, `golang.org/x/image` (new; pure-Go webp/bmp decode + scaling), fuzzel (dmenu + Rofi icon protocol), POSIX sh, `wl-clipboard`.

## Global Constraints

- **Two design specs back this plan:** `docs/superpowers/specs/2026-06-25-copyzen-image-previews-design.md` and `…-copyzen-delete-clear-design.md`. Both ship as one **v0.2.0**.
- **Static binary:** `CGO_ENABLED=0` must keep building; only pure-Go module deps allowed. No new *runtime* dependencies.
- **No store-schema change.** Detect images by sniffing magic bytes, not by persisting a MIME.
- **Keybindings are additive — never override a fuzzel default.** Existing: `custom-1=Control+s` (pin). New: `custom-2=Control+d` (delete), `custom-3=Control+Shift+c` (clear). fuzzel custom-N exit codes: custom-1→10, custom-2→11, custom-3→12. (`Control+Shift+c` must be lowercase `c` — fuzzel never fires upper-case Shift combos.)
- **Confirmations are fail-safe:** in a confirm state only the exact same key confirms; Enter/Esc/anything else cancels with no destructive action.
- **Thumbnails:** longest side ≤ 128px, re-encoded PNG, cached at `$XDG_CACHE_HOME/copyzen/thumbs/<id>.png` (default `~/.cache/copyzen/thumbs/`). Generated lazily on cache-miss; bytes fetched only on miss; orphans pruned. A cache/decode failure must never break the picker.
- **Image formats:** PNG, JPEG, GIF, WebP, BMP.
- **Commits:** author is `Harduex` with **no Claude co-author trailer** (repo-local git config already enforces this — use plain `git commit`).
- **Verification:** unit tests via `go test ./...`; the menu wrapper is POSIX sh and is verified by **running on the real machine** (this project's bugs only ever surfaced by running).

```go
package store

import (
	"bytes"
	"fmt"
	"strconv"
	"strings"
)

// SniffImage reports the image MIME type and file extension of b by inspecting its
// magic bytes. ok is false for any payload that is not one of the supported formats.
func SniffImage(b []byte) (mime, ext string, ok bool) {
	switch {
	case len(b) >= 8 && bytes.HasPrefix(b, []byte{0x89, 'P', 'N', 'G', 0x0D, 0x0A, 0x1A, 0x0A}):
		return "image/png", "png", true
	case len(b) >= 3 && bytes.HasPrefix(b, []byte{0xFF, 0xD8, 0xFF}):
		return "image/jpeg", "jpg", true
	case len(b) >= 6 && (bytes.HasPrefix(b, []byte("GIF87a")) || bytes.HasPrefix(b, []byte("GIF89a"))):
		return "image/gif", "gif", true
	case len(b) >= 12 && bytes.HasPrefix(b, []byte("RIFF")) && bytes.Equal(b[8:12], []byte("WEBP")):
		return "image/webp", "webp", true
	case len(b) >= 2 && bytes.HasPrefix(b, []byte("BM")):
		return "image/bmp", "bmp", true
	}
	return "", "", false
}

// ImageLabel renders the DISPLAY preview for an image entry, e.g. "🖼 PNG · 2.3 MB".
func ImageLabel(mime string, n int) string {
	return "🖼 " + strings.ToUpper(strings.TrimPrefix(mime, "image/")) + " · " + humanSize(n)
}

// humanSize formats a byte count as "512 B", "2.0 KB", "2.3 MB", etc.
func humanSize(n int) string {
	const unit = 1024
	if n < unit {
		return strconv.Itoa(n) + " B"
	}
	div, exp := int64(unit), 0
	for x := int64(n) / unit; x >= unit; x /= unit {
		div *= unit
		exp++
	}
	return fmt.Sprintf("%.1f %cB", float64(n)/float64(div), "KMGT"[exp])
}
```

```go
// ThumbMax is the longest-side pixel cap for cached thumbnails.
const ThumbMax = 128

// Thumbnail decodes src (png/jpeg/gif/webp/bmp), downscales it so the longest side is
// at most maxPx (never upscaling, aspect preserved), and re-encodes it as PNG.
func Thumbnail(src []byte, maxPx int) ([]byte, error) {
	img, _, err := image.Decode(bytes.NewReader(src))
	if err != nil {
		return nil, err
	}
	b := img.Bounds()
	w, h := b.Dx(), b.Dy()
	if w == 0 || h == 0 {
		return nil, errors.New("copyzen: empty image")
	}
	tw, th := w, h
	if w > maxPx || h > maxPx {
		if w >= h {
			tw, th = maxPx, h*maxPx/w
		} else {
			tw, th = w*maxPx/h, maxPx
		}
		if tw < 1 {
			tw = 1
		}
		if th < 1 {
			th = 1
		}
	}
	dst := image.NewRGBA(image.Rect(0, 0, tw, th))
	draw.ApproxBiLinear.Scale(dst, dst.Bounds(), img, b, draw.Over, nil)
	var out bytes.Buffer
	if err := png.Encode(&out, dst); err != nil {
		return nil, err
	}
	return out.Bytes(), nil
}
```

```go
// ThumbDir is $XDG_CACHE_HOME/copyzen/thumbs, falling back to ~/.cache.
func ThumbDir() (string, error) {
	dir := os.Getenv("XDG_CACHE_HOME")
	if dir == "" {
		home, err := os.UserHomeDir()
		if err != nil {
			return "", err
		}
		dir = filepath.Join(home, ".cache")
	}
	return filepath.Join(dir, "copyzen", "thumbs"), nil
}

// ThumbPath is the cache file for id's thumbnail.
func ThumbPath(id uint64) (string, error) {
	dir, err := ThumbDir()
	if err != nil {
		return "", err
	}
	return filepath.Join(dir, strconv.FormatUint(id, 10)+".png"), nil
}

// EnsureThumb returns the cached thumbnail path for id, generating it if absent. src is
// invoked only on a cache miss, so callers avoid loading payload bytes for cache hits.
func EnsureThumb(id uint64, maxPx int, src func() ([]byte, error)) (string, error) {
	path, err := ThumbPath(id)
	if err != nil {
		return "", err
	}
	if _, err := os.Stat(path); err == nil {
		return path, nil
	}
	data, err := src()
	if err != nil {
		return "", err
	}
	thumb, err := Thumbnail(data, maxPx)
	if err != nil {
		return "", err
	}
	if err := os.MkdirAll(filepath.Dir(path), 0o700); err != nil {
		return "", err
	}
	if err := os.WriteFile(path, thumb, 0o600); err != nil {
		return "", err
	}
	return path, nil
}

// PruneThumbs removes cached thumbnails whose id is not in keep. Best-effort: a missing
// cache dir is not an error, and per-file removal errors are ignored.
func PruneThumbs(keep map[uint64]bool) error {
	dir, err := ThumbDir()
	if err != nil {
		return err
	}
	ents, err := os.ReadDir(dir)
	if err != nil {
		if os.IsNotExist(err) {
			return nil
		}
		return err
	}
	for _, e := range ents {
		name := e.Name()
		if !strings.HasSuffix(name, ".png") {
			continue
		}
		id, perr := strconv.ParseUint(strings.TrimSuffix(name, ".png"), 10, 64)
		if perr != nil || keep[id] {
			continue
		}
		_ = os.Remove(filepath.Join(dir, name))
	}
	return nil
}
```

**Files:**
- Modify: `scripts/copyzen-menu`

**Interfaces:**
- Consumes: `copyzen list|decode|toggle|delete|wipe|mimetype`; fuzzel exit codes 0/10/11/12; `dist/fuzzel.ini` bindings (Task 8); `wl-copy`.

This task's deliverable is shell control flow that fuzzel exits cannot unit-test; it is verified by **running on the real machine**. Implement the script, then run the manual checklist.

- [ ] **Step 1: Rewrite the wrapper**

Replace the entire contents of `scripts/copyzen-menu` with:
```sh
#!/bin/sh
# copyzen-menu — fuzzel-based clipboard picker for copyzen. Bind to Super+V.
#   Enter          copy selection (type-aware: images paste back as images)
#   Ctrl+S         pin/unpin (toggle) selection
#   Ctrl+D         delete selection      (confirm: press Ctrl+D again, Esc cancels)
#   Ctrl+Shift+C   clear ALL history     (confirm: press Ctrl+Shift+C again, Esc cancels)
#   Esc            quit
# fuzzel exits on every action key and cannot reload its list, so each action
# relaunches fuzzel; confirmations relaunch into a confirm-state view. Pinned items
# are never cleared by Ctrl+Shift+C — remove them one-by-one with Ctrl+D.
# No `set -e`: fuzzel's non-zero exits are control flow, not errors.
set -u

config="${XDG_CONFIG_HOME:-$HOME/.config}/copyzen/fuzzel.ini"

# Base args for every fuzzel call. --with-nth (hides the id column) needs a newer
# fuzzel and is purely cosmetic, so it is added only to the main list, not the
# single-line confirm views (where there is no id column to hide).
set -- --dmenu --config "$config"
withnth=
if fuzzel --help 2>&1 | grep -q -- '--with-nth'; then
	withnth=--with-nth=2
fi

state=list
sel=

while :; do
	case "$state" in
	list)
		sel=$(copyzen list | fuzzel "$@" $withnth)
		case "$?" in
		0)  if [ -n "$sel" ]; then
				t=$(printf '%s' "$sel" | copyzen mimetype)
				if [ -n "$t" ]; then
					printf '%s' "$sel" | copyzen decode | wl-copy --type "$t"
				else
					printf '%s' "$sel" | copyzen decode | wl-copy
				fi
			fi
			exit 0 ;;
		10) [ -n "$sel" ] && printf '%s' "$sel" | copyzen toggle ;;
		11) [ -n "$sel" ] && state=confirm-delete ;;
		12) state=confirm-clear ;;
		*)  exit 0 ;;
		esac
		;;
	confirm-delete)
		# Show only the targeted entry's preview (strip "id<TAB>"). Only Ctrl+D (11)
		# confirms; Enter/Esc/anything else cancels. Deletes the captured id.
		preview=$(printf '%s' "$sel" | sed 's/^[0-9]*	//')
		printf '%s\n' "$preview" | fuzzel "$@" \
			--prompt '🗑 Delete this entry?  Ctrl+D = confirm · Esc = cancel ' >/dev/null
		[ "$?" -eq 11 ] && printf '%s' "$sel" | copyzen delete
		state=list
		;;
	confirm-clear)
		# Hide the list, show counts + a warning. Only Ctrl+Shift+C (12) confirms.
		counts=$(copyzen list | tr -d '\0' | awk -F'	' '
			{ total++; if ($2 ~ /^★/) pinned++ }
			END { printf "%d history items · %d pinned kept", total - pinned, pinned }')
		printf 'Clear %s\n' "$counts" | fuzzel "$@" \
			--prompt '⚠ Clear ALL history?  Ctrl+Shift+C = confirm · Esc = cancel ' >/dev/null
		[ "$?" -eq 12 ] && copyzen wipe
		state=list
		;;
	esac
done
```
**Note:** the `sed` and `awk` field separators above are literal TAB characters — when editing, ensure they are real tabs (`s/^[0-9]*<TAB>//` and `-F'<TAB>'`), not spaces.
